Train a classic model
----------------------------------------------------------------------------

    model = NeuralNetClassifier(
        MNISTtoUSPSNet(),
        criterion=nn.CrossEntropyLoss(),
        batch_size=128,
        max_epochs=5,
        train_split=False,
        lr=1e-2,

    model.fit(X[sample_domain > 0], y[sample_domain > 0])

        layer_name="fc1",
        batch_size=128,
        max_epochs=5,

        lr=1e-2,
